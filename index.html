import requests from bs4 import BeautifulSoup from datetime import datetime def scrape_bloomberg_headlines(): url = "https://www.bloomberg.co.jp" # Custom headers are crucial to avoid being blocked immediately headers = { "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36", "Accept-Language": "ja,en-US;q=0.9,en;q=0.8", "Referer": "https://www.google.com/" } try: print(f"Fetching headlines from {url}...") response = requests.get(url, headers=headers, timeout=15) response.raise_for_status() soup = BeautifulSoup(response.text, 'html.parser') # Bloomberg's structure often uses tags with specific class patterns # or story-package components. We target the common headline link pattern. articles = [] # This selector targets most main headlines on the homepage for link in soup.select('a[class*="story-package"], a[class*="headline"]'): title = link.get_text(strip=True) href = link.get('href') # Filter out short fragments or empty links if title and href and len(title) > 5: # Ensure absolute URL full_url = href if href.startswith('http') else f"https://www.bloomberg.co.jp{href}" # Avoid duplicates if {"title": title, "url": full_url} not in articles: articles.append({"title": title, "url": full_url}) if not articles: print("No articles found. The site structure might have changed.") return generate_html(articles) print(f"Successfully scraped {len(articles)} headlines. saved to 'headlines.html'.") except requests.exceptions.HTTPError as e: if e.response.status_code == 403: print("Access Denied (403). Bloomberg's bot protection is blocking the script.") else: print(f"HTTP Error: {e}") except Exception as e: print(f"An error occurred: {e}") def generate_html(articles): now = datetime.now().strftime("%Y-%m-%d %H:%M") html_content = f""" Bloomberg.jp Headlines

Bloomberg.jp Headlines

Scraped on: {now}

    """ for article in articles: html_content += f'

""" with open("headlines.html", "w", encoding="utf-8") as f: f.write(html_content) if __name__ == "__main__": scrape_bloomberg_headlines()
